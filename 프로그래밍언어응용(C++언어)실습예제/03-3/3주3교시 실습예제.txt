//실습예제 1-4 ====================================================

#include <iostream>
using namespace std;

int main(int argc, char **argv)
{
    for(int i = 0; i < argc; i++)
       cout << "argument " << i << ": " << argv[i] << endl;
    system("pause");
    return 0;
}

{

   return  0;
}
